function getWeekRange(dateKey) {
  const date = new Date(`${dateKey}T12:00:00`);
  const day = date.getDay();
  const diffToMonday = day === 0 ? -6 : 1 - day;

  const monday = new Date(date);
  monday.setDate(date.getDate() + diffToMonday);

  const sunday = new Date(monday);
  sunday.setDate(monday.getDate() + 6);

  return {
    start: dateKeyLocal(monday),
    end: dateKeyLocal(sunday),
  };
}

function portionLevel(portions) {
  if (portions >= 10) return { label: 'rosso', bg: 'bg-red-200', text: 'text-red-800', color: '#fecaca' };
  if (portions >= 7) return { label: 'arancio', bg: 'bg-orange-200', text: 'text-orange-800', color: '#fed7aa' };
  if (portions >= 4) return { label: 'verde', bg: 'bg-green-200', text: 'text-green-800', color: '#bbf7d0' };
  return { label: 'bianco', bg: 'bg-white', text: 'text-slate-700', color: '#ffffff' };
}

const weeklyIngredientUsage = useMemo(() => {
  const { start, end } = getWeekRange(selectedDate);
  const gramsByFood = {};

  Object.entries(days).forEach(([date, entries]) => {
    if (date < start || date > end) return;

    entries.forEach((entry) => {
      const food = foods.find((f) => f.id === entry.foodId);
      if (!food) return;

      if (food.isRecipe && Array.isArray(food.ingredients)) {
        const factor = (Number(entry.grams) || 0) / (food.recipeTotalWeight || 100);

        food.ingredients.forEach((ing) => {
          gramsByFood[ing.foodId] = (gramsByFood[ing.foodId] || 0) + ing.grams * factor;
        });
      } else {
        gramsByFood[food.id] = (gramsByFood[food.id] || 0) + (Number(entry.grams) || 0);
      }
    });
  });

  const result = {};

  foods.forEach((food) => {
    const grams = gramsByFood[food.id] || 0;
    const portionSize = Number(food.portionSize) || 100;
    const portions = grams / portionSize;

    result[food.id] = {
      grams: round1(grams),
      portions: round1(portions),
      level: portionLevel(portions),
    };
  });

  return result;
}, [days, foods, selectedDate]);

function recipeCircleStyle(recipe) {
  if (!recipe?.isRecipe || !Array.isArray(recipe.ingredients)) return null;

  const totalRecipeWeight = recipe.recipeTotalWeight || recipe.ingredients.reduce((s, i) => s + i.grams, 0) || 1;

  const parts = {
    red: 0,
    orange: 0,
    green: 0,
    white: 0,
  };

  recipe.ingredients.forEach((ing) => {
    const share = ing.grams / totalRecipeWeight;
    const usage = weeklyIngredientUsage[ing.foodId];
    const level = usage?.level?.label || 'bianco';

    if (level === 'rosso') parts.red += share;
    else if (level === 'arancio') parts.orange += share;
    else if (level === 'verde') parts.green += share;
    else parts.white += share;
  });

  const red = Math.round(parts.red * 100);
  const orange = Math.round(parts.orange * 100);
  const green = Math.round(parts.green * 100);
  const white = Math.max(0, 100 - red - orange - green);

  let start = 0;
  const segs = [];

  if (red) {
    segs.push(`#fecaca ${start}% ${start + red}%`);
    start += red;
  }
  if (orange) {
    segs.push(`#fed7aa ${start}% ${start + orange}%`);
    start += orange;
  }
  if (green) {
    segs.push(`#bbf7d0 ${start}% ${start + green}%`);
    start += green;
  }
  if (white) {
    segs.push(`#ffffff ${start}% 100%`);
  }

  return {
    background: `conic-gradient(${segs.join(', ')})`,
  };
}
